#include <LiquidCrystal.h>

// Pin Definitions #define trigger 10 #define echo 11 #define motor 8
#define buzzer 12

// Initialize LCD (RS, E, D4, D5, D6, D7) LiquidCrystal lcd(7, 6, 5, 4,
3, 2);

// Variables float time = 0; float distance = 0; int temp = 0;

void setup() { lcd.begin(16, 2);

pinMode(trigger, OUTPUT); pinMode(echo, INPUT); pinMode(motor, OUTPUT);
pinMode(buzzer, OUTPUT);

lcd.print(” Water Level “); lcd.setCursor(0, 1); lcd.print(” Indicator
“); delay(2000); }

void loop() { lcd.clear();

// Trigger ultrasonic pulse digitalWrite(trigger, LOW);
delayMicroseconds(2); digitalWrite(trigger, HIGH);
delayMicroseconds(10); digitalWrite(trigger, LOW);

// Read echo time time = pulseIn(echo, HIGH);

// Calculate distance (in cm) distance = (time * 0.0343) / 2;

// Display distance lcd.clear(); lcd.print(“Water Space In”);
lcd.setCursor(0, 1); lcd.print(“Tank:”); lcd.print(distance);
lcd.print(” cm”);

delay(2000);

// Tank Full Condition if (distance < 12 && temp == 0) {
digitalWrite(motor, LOW); // Motor OFF digitalWrite(buzzer, HIGH); //
Buzzer ON

    lcd.clear();
    lcd.print("Tank Full");
    lcd.setCursor(0, 1);
    lcd.print("Motor OFF");

    delay(2000);
    digitalWrite(buzzer, LOW);
    delay(3000);

    temp = 1;

}

else if (distance < 12 && temp == 1) { digitalWrite(motor, LOW);

    lcd.clear();
    lcd.print("Tank Full");
    lcd.setCursor(0, 1);
    lcd.print("Motor OFF");

    delay(3000);

}

// Low Water Level Condition else if (distance > 30) {
digitalWrite(motor, HIGH); // Motor ON

    lcd.clear();
    lcd.print("Low Water Level");
    lcd.setCursor(0, 1);
    lcd.print("Motor ON");

    delay(3000);

    temp = 0;

} }
